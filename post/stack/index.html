// Package stack implements a basic stack data structure.
package stack



// Stack holds a reference to the top of the stack
// and the current size.
type Stack struct {
    top  *Item
    size int
}



// Item holds the value for the item and
// the next Item.
type Item struct {
    value interface{}
    next  *Item
}



// Len returns the stacks length.
func (s *Stack) Len() int {
    return s.size
}



// Push adds a new element onto the stack and
// sets top to the pushed item.
func (s *Stack) Push(value interface{}) {
    s.top = &Item{value: value, next: s.top}
    s.size++
}



// Pop removes the top element from the stack and returns the value
// along with setting top to the next element.
// If the stack is empty, return nil
func (s *Stack) Pop() interface{} {
    if s.size > 0 {
        value := s.top.value
        s.top = s.top.next
        s.size–
        return value
    }
    return nil
}



// New is a helper for an initial stack.
func New() *Stack {
    return &Stack{}
}
